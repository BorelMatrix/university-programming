1function [A,b,x]=Gaussel(A,b,tol)
 2[m,n] = size(A);
 3% Compute the default tolerance if none was provided.
 4if (nargin < 3), tol = max(m,n)*eps(class(A))*norm(A,'inf'); end
 5% Loop over the entire matrix.
 6i = 1;
 7j = 1;
 8while (i <= m) && (j <= n)
 9    % Find value and index of largest element in the remainder of column j.
10    [p,k] = max(abs(A(i:m,j))); k = k+i-1;
11    % k referd the index.
12    if (p <= tol)
13        % The column is negligible,end the function;
14        break;
15    else
16        % Swap i-th and k-th rows.
17        A([i k],j:n) = A([k i],j:n);
18        b([i k])=b([k i]);
19        % Divide the pivot row by the pivot element.
20        b(i)=b(i)/A(i,j);
21        A(i,j:n) = A(i,j:n)/A(i,j);
22        % Subtract multiples of the pivot row from all the other rows.
23        for k = i+1:m
24            b(k)=b(k)-A(k,j)*b(i);
25            A(k,j:n) = A(k,j:n) - A(k,j)*A(i,j:n);
26        end
27        i = i + 1;
28        j = j + 1;
29    end
30end
31x=b;
32% calculate x.
33for i=n-1:-1:1
34    x(i)=x(i)-A(i,i+1:n)*x(i+1:n);
35end
